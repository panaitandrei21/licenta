select a.ename, b.dname, a.sal + nvl(a.comm, 0) venit, 
case
	when (a.sal + nvl(a.comm, 0)) < 1500 then 'slabut'
	when (a.sal + nvl(a.comm, 0)) > 1500 and (a.sal + nvl(a.comm, 0)) < 2500 then 'ok'
	else 'top'
end cume
from
emp a join dept b on a.deptno = b.deptno;


select a.ename nume_ang, s.ename nume_sef, g.grade,
case
	when g.grade < 3 then 'scazut'
	when g.grade >= 3 and g.grade <5 then 'mediu'
	else 'ridicat'
end apreciere
from emp a join emp s on a.mgr = s.empno
	join salgrade g on (a.sal >= g.losal and a.sal < g.hisal);

select d.dname, d.loc, count(a.empno) nr
from emp a join dept d on a.deptno = d.deptno
where nvl(a.comm, 0) = 0
group by d.dname, d.loc;

select g.grade, count(a.empno)
from emp a join salgrade g on a.sal > g.losal and a.sal < g.hisal
group by g.grade;
